	venta_id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	cliente_id INTEGER,
	producto_id INTEGER,
	tiempo_id NUMBER,
	cantidad REAL,
	valor_total REAL,
	canal_id INTEGER,

	CONSTRAINT FACT_venta_DIM_producto_FK FOREIGN KEY (producto_id) REFERENCES DIM_producto(producto_id),
